    <script>
        // --- Three.js Background (Layer 2) ---
        let scene, camera, renderer, hearts = [];
        const container = document.getElementById('canvas-container');

        function initThree() {
            scene = new THREE.Scene();
            camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
            renderer = new THREE.WebGLRenderer({ alpha: true, antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            container.appendChild(renderer.domElement);

            const ambientLight = new THREE.AmbientLight(0xffffff, 0.5);
            scene.add(ambientLight);
            const pointLight = new THREE.PointLight(0xffffff, 1);
            pointLight.position.set(5, 5, 5);
            scene.add(pointLight);

            // Create Heart Shape
            const shape = new THREE.Shape();
            const x = 0, y = 0;
            shape.moveTo(x + 5, y + 5);
            shape.bezierCurveTo(x + 5, y + 5, x + 4, y, x, y);
            shape.bezierCurveTo(x - 6, y, x - 6, y + 7, x - 6, y + 7);
            shape.bezierCurveTo(x - 6, y + 11, x - 3, y + 15.4, x + 5, y + 19);
            shape.bezierCurveTo(x + 12, y + 15.4, x + 16, y + 11, x + 16, y + 7);
            shape.bezierCurveTo(x + 16, y + 7, x + 16, y, x + 10, y);
            shape.bezierCurveTo(x + 7, y, x + 5, y + 5, x + 5, y + 5);

            const geometry = new THREE.ExtrudeGeometry(shape, { depth: 2, bevelEnabled: true, bevelThickness: 1, bevelSize: 1 });
            geometry.center();

            for(let i=0; i<25; i++) {
                const material = new THREE.MeshStandardMaterial({ 
                    color: new THREE.Color(`hsl(${Math.random() * 20 + 340}, 80%, 60%)`),
                    roughness: 0.3,
                    metalness: 0.2
                });
                const heart = new THREE.Mesh(geometry, material);
                heart.position.set(
                    (Math.random() - 0.5) * 100,
                    (Math.random() - 0.5) * 100,
                    (Math.random() - 0.5) * 50
                );
                heart.rotation.x = Math.PI;
                heart.rotation.z = Math.random() * Math.PI;
                heart.scale.set(0.2, 0.2, 0.2);
                
                // Add custom data for movement
                heart.userData = {
                    speed: 0.01 + Math.random() * 0.02,
                    angle: Math.random() * Math.PI * 2
                };
                
                hearts.push(heart);
                scene.add(heart);
            }
            camera.position.z = 50;
        }

        function animateThree() {
            requestAnimationFrame(animateThree);
            hearts.forEach(heart => {
                heart.rotation.y += 0.01;
                heart.position.y += Math.sin(heart.userData.angle) * 0.05;
                heart.userData.angle += heart.userData.speed;
            });
            renderer.render(scene, camera);
        }

        // --- Step Logic ---
        let currentStep = 1;
        const totalSteps = 5;

        function nextStep(step) {
            const prevEl = document.getElementById(`step${currentStep}`);
            const nextEl = document.getElementById(`step${step}`);
            
            gsap.to(prevEl, {
                opacity: 0,
                y: -20,
                scale: 0.95,
                duration: 0.5,
                onComplete: () => {
                    prevEl.style.display = 'none';
                    nextEl.style.display = 'block';
                    gsap.fromTo(nextEl, 
                        { opacity: 0, y: 20, scale: 1.05 },
                        { opacity: 1, y: 0, scale: 1, duration: 0.6, ease: "power2.out" }
                    );
                    
                    // Stagger child elements
                    gsap.from(nextEl.children, {
                        opacity: 0,
                        y: 10,
                        stagger: 0.1,
                        duration: 0.5,
                        delay: 0.2
                    });
                }
            });

            currentStep = step;
            document.getElementById('progressBar').style.width = `${(step/totalSteps) * 100}%`;
        }

        // --- Polaroid Parallax ---
        const polaroid = document.getElementById('polaroid');
        document.addEventListener('mousemove', (e) => {
            if (currentStep !== 4) return;
            const x = (window.innerWidth / 2 - e.pageX) / 25;
            const y = (window.innerHeight / 2 - e.pageY) / 25;
            polaroid.style.transform = `rotateY(${x}deg) rotateX(${y}deg) rotate(-3deg)`;
        });

        // --- Finale ---
        function grandFinale() {
            const btn = document.getElementById('celebrateBtn');
            const finalMsg = document.getElementById('finalMessage');
            
            btn.style.pointerEvents = 'none';
            gsap.to(btn, { opacity: 0, y: 10, duration: 0.5 });
            
            setTimeout(() => {
                finalMsg.classList.remove('hidden');
                gsap.fromTo(finalMsg, 
                    { opacity: 0, y: 20 },
                    { opacity: 1, y: 0, duration: 1, ease: "back.out" }
                );
            }, 500);

            // Confetti Explosion
            const duration = 5 * 1000;
            const animationEnd = Date.now() + duration;
            const defaults = { startVelocity: 30, spread: 360, ticks: 60, zIndex: 0 };

            function randomInRange(min, max) {
                return Math.random() * (max - min) + min;
            }

            const interval = setInterval(function() {
                const timeLeft = animationEnd - Date.now();
                if (timeLeft <= 0) return clearInterval(interval);

                const particleCount = 50 * (timeLeft / duration);
                confetti({...defaults, particleCount, origin: { x: randomInRange(0.1, 0.3), y: Math.random() - 0.2 }});
                confetti({...defaults, particleCount, origin: { x: randomInRange(0.7, 0.9), y: Math.random() - 0.2 }});
            }, 250);

            // Emoji Confetti
            const scalar = 2;
            const heartIcon = confetti.shapeFromText({ text: '‚ù§Ô∏è', scalar });
            const starIcon = confetti.shapeFromText({ text: '‚ú®', scalar });

            setTimeout(() => {
                confetti({
                    shapes: [heartIcon, starIcon],
                    particleCount: 40,
                    spread: 70,
                    origin: { y: 0.6 }
                });
            }, 1000);

            // 3D Hearts Fly Away
            hearts.forEach((heart, i) => {
                gsap.to(heart.position, {
                    x: (Math.random() - 0.5) * 200,
                    y: 100,
                    z: (Math.random() - 0.5) * 200,
                    duration: 3 + Math.random() * 2,
                    delay: i * 0.02,
                    ease: "power2.in"
                });
                gsap.to(heart.material, {
                    opacity: 0,
                    duration: 2,
                    delay: 2 + i * 0.02,
                    transparent: true
                });
            });
        }

        // Initialize
        window.addEventListener('load', () => {
            initThree();
            animateThree();
            document.getElementById('progressBar').style.width = '20%';
            
            // Fade in first card
            gsap.fromTo("#step1", 
                { opacity: 0, y: 30 },
                { opacity: 1, y: 0, duration: 1.2, ease: "power3.out" }
            );
        });

        // Handle Resize
        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });
    </script>
